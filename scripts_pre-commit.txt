# Installation pre-commit (via uv)
uv tool install pre-commit

# Nettoyage configuration
uvx pre-commit clean

# Installation des hooks
uvx pre-commit install --install-hooks

# Tous les hooks sur tous les fichiers
uvx pre-commit run --files @(Get-ChildItem -Recurse src,tests -Filter *.py | ForEach-Object { $_.FullName })

uvx pre-commit run --all-files

# PyTest Coverage
uv run pytest --cov=src --cov-report=term-missing --cov-report=html
start htmlcov/index.html
